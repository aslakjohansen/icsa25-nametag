\newcommand{\border}{8mm}

\newcommand{\TopBanner}[2]{
  \begin{tikzpicture}[remember picture,overlay]
    \node[
      anchor=north west,
      fill=#1,
      text=white,
      minimum width=\paperwidth,
      minimum height=24mm,
    ] at ([yshift=-15mm] current page.north west) {#2};
  \end{tikzpicture}
}

\newcommand{\MainText}[2]{
  \begin{tikzpicture}[remember picture,overlay]
    \node[
      overlay,
      anchor=north west,
      text=#1,
      fill=lime, % for debugging
      align=left,
      minimum width=\paperwidth-2*\border,
    ] at ([xshift=\border,yshift=-15mm-24mm-\border] current page.north west) {
      \begin{minipage}{85mm}
        \raggedright\huge
        \begin{bfseries}
          \sffamily
          #2
        \end{bfseries}
      \end{minipage}
    };
  \end{tikzpicture}
}

\newcommand{\BottomBanner}[2]{
  \begin{tikzpicture}[remember picture,overlay]
    \node[
      anchor=south west,
      fill=#1,
      text=white,
      minimum width=\paperwidth,
      minimum height=16mm,
    ] at ([yshift=12mm] current page.south west) {\Huge{\textbf{#2}}};
  \end{tikzpicture}
}
\newcommand{\OptionBanner}[1]{
  \begin{tikzpicture}[remember picture,overlay]
    \node[
      anchor=south west,
      fill=#1!20,
      text=white,
      minimum width=\paperwidth,
      minimum height=12mm,
    ] at (current page.south west) {};
  \end{tikzpicture}
}
