\newcommand{\TopBanner}[2]{
  \begin{tikzpicture}[remember picture,overlay]
    \node[
      anchor=north west,
      fill=#1,
      text=white,
      minimum width=\paperwidth,
      minimum height=24mm,
    ] at ([yshift=-15mm] current page.north west) {
      \Huge\sffamily
      \begin{bfseries}
        \scalebox{1.4}{#2}
      \end{bfseries}
    };
  \end{tikzpicture}
}

\newcommand{\MainText}[2]{
  \begin{tikzpicture}[remember picture,overlay]
    \node[
      overlay,
      anchor=north west,
      text=#1,
%      fill=lime, % for debugging
      align=left,
      minimum width=\paperwidth-2*\border,
    ] at ([xshift=\border,yshift=-15mm-24mm-\border] current page.north west) {
      \begin{minipage}{85mm}
        \raggedright\huge
        \begin{bfseries}
          \sffamily
          #2
        \end{bfseries}
      \end{minipage}
    };
  \end{tikzpicture}
}
